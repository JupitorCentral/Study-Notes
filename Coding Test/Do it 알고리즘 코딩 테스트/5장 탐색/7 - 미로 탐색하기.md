``` java

public class App {

    static StringTokenizer st;
    static StringBuilder sb;
    static int[][] G;
    static boolean[][] visited;
    static Point dest;

    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        sb = new StringBuilder();

        st = new StringTokenizer(br.readLine());
        int n = Integer.parseInt(st.nextToken());
        int m = Integer.parseInt(st.nextToken());

        G = new int[n][m];
        for (int i=0; i<n; i++) {
            char[] chars = br.readLine().toCharArray();
            for (int j=0; j<m; j++) {
                G[i][j] = Integer.parseInt(String.valueOf(chars[j]));
            }
        }
        visited = new boolean[n][m];

        Stack<Point> stack = new Stack<>();
        stack.push(new Point(0, 0, 1));
        while (!stack.isEmpty()) {
            Point current = stack.pop();
            visited[current.x][current.y] = true;
            int cx = current.x;
            int cy = current.y;
            int cc = current.count;
            boolean available = false;

            // decision
            if (cx == n-1 && cy == m-1) {
                if (dest == null || dest.count > cc) dest = current;
                visited[cx][cy] = false;
                continue;
            }

            // left
            if (cy - 1 >= 0 && G[cx][cy-1] == 1 && !visited[cx][cy-1]) {
                stack.add(new Point(cx, cy - 1, cc + 1));
                available = true;
            }

            // up
            if (cx-1 >=0 && G[cx-1][cy] == 1 && !visited[cx-1][cy]) {
                stack.add(new Point(cx-1, cy, cc+1));
                available = true;
            }

            // right
            if (cy + 1 < m && G[cx][cy+1] == 1 && !visited[cx][cy+1]) {
                stack.add(new Point(cx, cy + 1, cc + 1));
                available = true;
            }
            // down
            if (cx + 1 < n && G[cx+1][cy] == 1 && !visited[cx+1][cy]) {
                stack.add(new Point(cx+1, cy, cc + 1));
                available = true;
            }

            if (!available) visited[cx][cy] = false;
        }

        sb.append(dest.count);


        bw.write(sb.toString());
        bw.newLine();
        bw.flush();
        bw.close();
    }

    public static class Point {
        public int x; // 행
        public int y; // 렬
        public int count;
        Point(int x, int y, int count) {
            this.x = x;
            this.y = y;
            this.count = count;
        }
    }
}
```

``` java

public class App {

    static StringTokenizer st;
    static StringBuilder sb;
    static int[][] G;
    static boolean[][] visited;
    static int count;
    static int n;
    static int m;
    static int min;

    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        sb = new StringBuilder();

        st = new StringTokenizer(br.readLine());
        n = Integer.parseInt(st.nextToken());
        m = Integer.parseInt(st.nextToken());

        G = new int[n][m];
        for (int i=0; i<n; i++) {
            char[] chars = br.readLine().toCharArray();
            for (int j=0; j<m; j++) {
                G[i][j] = Integer.parseInt(String.valueOf(chars[j]));
            }
        }
        visited = new boolean[n][m];

        count = 0;
        min = Integer.MAX_VALUE;
        search(0, 0);

        sb.append(min);

        bw.write(sb.toString());
        bw.newLine();
        bw.flush();
        bw.close();
    }

    public static void search (int cx, int cy) {
        visited[cx][cy] = true;
        count++;

        // decision
        if (cx == n-1 && cy == m-1) {
            if (min > count) min = count;
            count--;
            visited[cx][cy] = false;
            return;
        }

        // left
        if (cy - 1 >= 0 && G[cx][cy-1] == 1 && !visited[cx][cy-1]) {
            search(cx, cy-1);
        }

        // up
        if (cx-1 >=0 && G[cx-1][cy] == 1 && !visited[cx-1][cy]) {
            search(cx-1, cy);
        }

        // right
        if (cy + 1 < m && G[cx][cy+1] == 1 && !visited[cx][cy+1]) {
            search(cx, cy+1);
        }
        // down
        if (cx + 1 < n && G[cx+1][cy] == 1 && !visited[cx+1][cy]) {
            search(cx+1, cy);
        }

        count--;
        visited[cx][cy] = false;
    }

    public static class Point {
        public int x; // 행
        public int y; // 렬
        public int count;
        Point(int x, int y, int count) {
            this.x = x;
            this.y = y;
            this.count = count;
        }
    }
}
```
