
확장 유클리즈 호제법을 이용하여 문제를 푼다.



``` java

public class App {
    static StringTokenizer st;
    static StringBuilder sb;
    static int c;
    static boolean noAnswer;

    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        sb = new StringBuilder();

        st = new StringTokenizer(br.readLine());
        int a = Integer.parseInt(st.nextToken());
        int b = Integer.parseInt(st.nextToken());
        c = Integer.parseInt(st.nextToken());

        int[] result = euclid(a, b); // result[0] = x, result[1] = y
        int x = result[0];
        int y = result[1];
        int cc = (c / (a*x + b*y));
        int xx = x * cc;
        int yy = y * cc;
        if (noAnswer) {
            sb.append(-1);
        } else {
            sb.append(xx).append(" ").append(yy);
        }

        bw.write(sb.toString());
        bw.newLine();
        bw.flush();
        bw.close();
    }

    public static int[] euclid (int a, int b) {
        if (b == 0) {
            if (c % a != 0) noAnswer = true;
            return new int[]{1, 0};
        }
        int R = a % b;
        int Q = a / b;
        int [] xy = euclid(b, R);
        int x = xy[0];
        int y = xy[1];

        int[] result = new int[]{0, 0};
        result[0] = y;
        result[1] = x - (y * Q);

        return result;
    }
}
```

euclid 에서 b = 0 일때 a 가 gcd 이고, <br>
이때 c 가 gcd  로 나눠지지 않으면 해가 존재하지 않기 때문에 이를 체크해주어야 한다.
